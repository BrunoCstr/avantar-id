rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Regras para a coleção de usuários
    match /users/{userId} {
      // Usuários podem ler/escrever seus próprios dados
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Administradores podem ler/escrever dados de todos os usuários
      allow read, write: if request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Regras para dados compartilhados de seguradoras
    match /global-companies/{document} {
      // Todos os usuários autenticados podem ler
      allow read: if request.auth != null;
      
      // Apenas administradores podem escrever
      allow write: if request.auth != null && 
                      exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
                      get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }
    
    // Regras para a coleção de companies por usuário
    match /user-companies/{userId} {
      // Usuários podem ler/escrever apenas seus próprios companies
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Admins podem ler/escrever companies de todos os usuários
      allow read, write: if request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Regras para logs de acesso
    match /access-logs/{logId} {
      // Qualquer usuário autenticado pode criar logs
      allow create: if request.auth != null;
      
      // Apenas administradores podem ler logs
      allow read: if request.auth != null && 
                     exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
                     get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
      
      // Ninguém pode atualizar ou deletar logs (imutáveis para auditoria)
      allow update, delete: if false;
    }
    
    // Negar acesso a qualquer outra coleção não especificada
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
